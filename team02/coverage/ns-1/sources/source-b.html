 package com.example.parkingspotmanager.domain;
 
 import java.util.ArrayList;
 import java.util.Scanner;
 
 public class RegistrationController {
     private static ArrayList<String> userDatabase = new ArrayList<>();
     private static ArrayList<String> emailInfo = new ArrayList<>();
     private static ArrayList<String> phoneInfo = new ArrayList<>();
     private static ArrayList<Registration> pendingRegistrations = new ArrayList<>();
     private static ArrayList<Registration> approvedRegistrations = new ArrayList<>();
 
     public Customer registerPersonalInfo(Scanner scanner){
         System.out.println("Please enter your First Name:");
         String firstName = scanner.nextLine();
         System.out.println("Please enter your Last Name:");
         String lastName = scanner.nextLine();
         System.out.println("Please enter your Address:");
         String address = scanner.nextLine();
         System.out.println("Please enter your Phone Number:");
         String phoneNumber = scanner.nextLine();
         System.out.println("Please enter your Email:");
         String email = scanner.nextLine();
         if (emailInfo.contains(email)){
             System.out.println("Email already exists.");
         } else if (phoneInfo.contains(phoneNumber)) {
             System.out.println("Phone number already exists.");
         } else {
             Customer tempcustomer = new Customer(firstName, lastName, address, phoneNumber, email);
             System.out.println("Personal information registered successfully.");
             return tempcustomer;
         }
         return null;
     }
 
     public Vehicle registerVehicleInfo(Customer customer, Scanner scanner){
         System.out.println("Please enter your Vehicle License Plate:");
         String licensePlate = scanner.nextLine();
         System.out.println("Please enter your Vehicle Brand:");
         String brand = scanner.nextLine();
         System.out.println("Please enter your Vehicle Model:");
         String model = scanner.nextLine();
         Vehicle vehicle = new Vehicle(licensePlate,brand, model);
         customer.setVehicle(vehicle);
         System.out.println("Vehicle information registered successfully.");
         return vehicle;
     }
 
     // Method to register a user. It checks if the username is already taken.
     public String registerUserName(Scanner scanner) {
         System.out.println("Please enter a username:");
         String username = scanner.nextLine();
         if (userDatabase.contains(username)) {
             System.out.println("Username already exists.");
             return null;
         } else {
             System.out.println("Username " + username + " is available. Please set the password.");
             userDatabase.add(username);
             return username;
         }
     }
 
     // Method to set password for a user. It is done separately from username registration to allow for password validation,
     // so that the user doesn't type the username again if the password is invalid.
     public String RegisterPassword(String username, Scanner scanner){
         System.out.println("Please enter a password:");
         String password = scanner.nextLine();
         if (password.length() < 8) {
             System.out.println("Password must be at least 8 characters long.");
         } else {
             //userDatabase.add(username, password);
             System.out.println("User " + username + " registered successfully.");
             return password;
         }
         return null;
     }
 
     public void register(){
 
         Scanner scanner = new Scanner(System.in);
         Customer tempCustomer = registerPersonalInfo(scanner);
         if (tempCustomer != null){
             registerVehicleInfo(tempCustomer,scanner);
             String username = registerUserName(scanner);
             if (username != null){
                 String password = RegisterPassword(username,scanner);
                 if (password != null) {
                     pendingRegistrations.add(new Registration(new Customer(tempCustomer.getEmail(), tempCustomer.getPhoneNumber(), username, password, tempCustomer.getVehicle())));
                 }
             }
         }
     }
 
     public ArrayList<String> getUserDatabase() {
         return userDatabase;
     }
 
     public ArrayList<String> getEmailInfo() {return emailInfo;}
 
     public ArrayList<String> getPhoneInfo() {return phoneInfo;}
 
     public ArrayList<Registration> getPendingRegistrations() {
         return pendingRegistrations;
     }
 
     public ArrayList<Registration> getApprovedRegistrations() {
         return approvedRegistrations;
     }
 
     public void reset(){
         userDatabase.clear();
         emailInfo.clear();
         phoneInfo.clear();
         pendingRegistrations.clear();
         approvedRegistrations.clear();
     }
 }

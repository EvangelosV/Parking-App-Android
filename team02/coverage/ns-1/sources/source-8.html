     public void processPayment(Customer customer) {
         if (getCustomer() != null) {
             if (getAmount() <= getCustomer().getAccountBalance()) {
                 getCustomer().setAccountBalance(customer.getAccountBalance() - amount);
                 setPaymentStatus(Status.APPROVED);
             } else {
                 setPaymentStatus(Status.REJECTED);
                 sendPaymentError(1);
             }
         }
         else {
             setPaymentStatus(Status.REJECTED);
             sendPaymentError(2);
         }
     }
 
     public void updatePaymentStatus(Status status) {
         if (status == Status.PENDING) {
             this.paymentStatus = Status.PENDING;
         } else if (status == Status.APPROVED) {
             this.paymentStatus = Status.APPROVED;
         } else if (status == Status.REJECTED) {
             this.paymentStatus = Status.REJECTED;
         }
     }
 
     public boolean sendPaymentConfirmation() {
         return this.paymentStatus == Status.APPROVED;
     }
 
     public void sendPaymentError(int errorCode) {
         switch (errorCode) {
             case 1:
                 System.out.print("Error: Insufficient funds\n");
                 break;
             case 2:
                 System.out.print("Error: Customer not found\n");
                 break;
         }
     }

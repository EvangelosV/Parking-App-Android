 package com.example.parkingspotmanager.domain;
 
 public class Employee extends User {
 
     RegistrationController registrationController = new RegistrationController();// Registration controller object
 
     QRCode QRCode = new QRCode();// QR code object
 
     // Constructor
     public Employee(String email, String phoneNumber, String username, String password) {
         super(email, phoneNumber, username, password);
     }
 
     // Method to approve a registration
     public void approveRegistration(Registration registration) {
         registration.setStatus(Status.APPROVED);
         System.out.println("Registration for " + registration.getCustomer().getUsername() + " approved by " + this.getUsername());
         QRCode.generateQRCode(registration.getCustomer(), registration.getCustomer().getUsername(), 350, 350);
         registrationController.getPendingRegistrations().remove(registration);
         registrationController.getApprovedRegistrations().add(registration);
         registrationController.getEmailInfo().add(registration.getCustomer().getEmail());
         registrationController.getPhoneInfo().add(registration.getCustomer().getPhoneNumber());
     }
 
     // Method to reject a registration
     public void rejectRegistration(Registration registration) {
         registration.setStatus(Status.REJECTED);
         System.out.println("Registration for " + registration.getCustomer().getUsername() + " rejected by " + this.getUsername());
         registrationController.getPendingRegistrations().remove(registration);
     }
 
 }    // Getter and Setter for QRCode

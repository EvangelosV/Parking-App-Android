 package com.example.parkingspotmanager.domain;
 
 import androidx.annotation.NonNull;
 
 import java.time.Duration;
 import java.time.LocalDateTime;
 import java.util.List;
 
 public class ParkingSpace {
     private String parkingSpaceId;// Parking Space ID
     private ParkingBuilding parkingBuilding;// Parking Building
     private boolean isOccupied;// Is the parking space occupied
     private boolean isReserved;// Is the parking space reserved
     private double pricePerHour;// Price per hour
 
     private List<Reservation> reservations;
 
     // Constructor
     public ParkingSpace(String parkingSpaceId, ParkingBuilding parkingBuilding, boolean isOccupied, boolean isReserved, double pricePerHour) {
         this.parkingSpaceId = parkingSpaceId;
         this.parkingBuilding = parkingBuilding;
         this.isOccupied = isOccupied;
         this.isReserved = isReserved;
         this.pricePerHour = pricePerHour;
     }
 
     // Getters and Setters
 
     public String getParkingSpaceId() {
         return parkingSpaceId;
     }
 
     public void setParkingSpaceId(String parkingSpaceId) {
         this.parkingSpaceId = parkingSpaceId;
     }
 
     public ParkingBuilding getParkingBuilding() {
         return parkingBuilding;
     }
 
     public void setParkingBuilding(ParkingBuilding parkingBuilding) {
         this.parkingBuilding = parkingBuilding;
     }
 
     public boolean isOccupied() {
         return isOccupied;
     }
 
     public void setOccupied(boolean occupied) {
         isOccupied = occupied;
     }
 
     public boolean isReserved() {
         return isReserved;

         isReserved = reserved;
     }
 
     public double getPricePerHour() {
         return pricePerHour;
     }
 
     public void setPricePerHour(double pricePerHour) {
         this.pricePerHour = pricePerHour;
     }
 
     // Method to calculate average daily occupancy
     public double calculateAverageDailyOccupancy() {
         // Assuming we have a method to get the total occupied time in hours
         double totalOccupiedHours = getTotalOccupiedHours();
 
         // Assuming we have a method to get the total days considered
         int totalDays = getTotalDaysConsidered();
 
         if (totalDays == 0) {
             return 0;
         }
 
         // Average daily occupancy is the total occupied hours divided by the total days
         return totalOccupiedHours / totalDays;
     }
 
     private double getTotalOccupiedHours() {
         return reservations.stream()
                 .mapToDouble(reservation -> Duration.between(reservation.getEntryTime(), reservation.getReservedExitTime()).toHours()).sum();
     }
 
     private int getTotalDaysConsidered() {
         if (reservations.isEmpty()) {
             return 0;
         }
 
         LocalDateTime firstReservation = reservations.stream()
                 .map(Reservation::getEntryTime)
                 .min(LocalDateTime::compareTo)
                 .orElse(LocalDateTime.now());
 
         LocalDateTime lastReservation = reservations.stream()
                 .map(Reservation::getReservedExitTime)
                 .max(LocalDateTime::compareTo)
                 .orElse(LocalDateTime.now());
 
         return (int) Duration.between(firstReservation, lastReservation).toDays() + 1;
     }
 
     public void setReservations(List<Reservation> list) {
         this.reservations = list;
     }
 }

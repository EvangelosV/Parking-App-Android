 package com.example.parkingspotmanager.domain;
 
 import android.graphics.Bitmap;
 import android.graphics.BitmapFactory;
 
 import com.fatboyindustrial.gsonjavatime.Converters;
 import com.google.gson.Gson;
 import com.google.gson.GsonBuilder;
 import com.google.zxing.BarcodeFormat;
 import com.google.zxing.BinaryBitmap;
 import com.google.zxing.ChecksumException;
 import com.google.zxing.FormatException;
 import com.google.zxing.RGBLuminanceSource;
 import com.google.zxing.MultiFormatReader;
 import com.google.zxing.NotFoundException;
 import com.google.zxing.Reader;
 import com.google.zxing.WriterException;
 import com.google.zxing.client.j2se.MatrixToImageWriter;
 import com.google.zxing.common.BitMatrix;
 import com.google.zxing.common.HybridBinarizer;
 import com.google.zxing.qrcode.QRCodeWriter;
 
 import java.io.File;
 import java.io.IOException;
 import java.nio.file.FileSystems;
 import java.nio.file.Path;
 
 public class QRCode {
     public static void generateQRCode(Object data, String username, int width, int height) {
         try {
             Gson gson = Converters.registerLocalDateTime(new GsonBuilder()).create();
             String dataString = gson.toJson(data);
 
             // Generate QR Code
             QRCodeWriter qrCodeWriter = new QRCodeWriter();
             BitMatrix bitMatrix = qrCodeWriter.encode(dataString, BarcodeFormat.QR_CODE, width, height);
 
             // Save QR Code
             String outputDirectory = "qrcodes/";
             saveQRCode(bitMatrix, outputDirectory + username + ".png");
 
         } catch (WriterException e) {
             System.err.println("Error generating QR Code: " + e.getMessage());
         }
     }
 
     public static void saveQRCode(BitMatrix bitMatrix, String filePath) {
         try {
             Path path = FileSystems.getDefault().getPath(filePath);
             MatrixToImageWriter.writeToPath(bitMatrix, "PNG", path);
         } catch (IOException e) {
             System.err.println("Error saving QR Code: " + e.getMessage());
         }
     }
 
     public static String readQRCode(String filePath) {
         try {
             // Read the image file
             File file = new File(filePath);
             Bitmap bitmap = BitmapFactory.decodeFile(file.getAbsolutePath());
 
             int[] intArray = new int[bitmap.getWidth() * bitmap.getHeight()];
             bitmap.getPixels(intArray, 0, bitmap.getWidth(), 0, 0, bitmap.getWidth(), bitmap.getHeight());
             RGBLuminanceSource source = new RGBLuminanceSource(bitmap.getWidth(), bitmap.getHeight(), intArray);
 
             // Create a BinaryBitmap from the LuminanceSource
             BinaryBitmap binaryBitmap = new BinaryBitmap(new HybridBinarizer(source));
 
             // Use ZXing's MultiFormatReader to decode the image
             Reader reader = new MultiFormatReader();
             return reader.decode(binaryBitmap).getText();
 
         } catch (NotFoundException e) {
             System.err.println("QR Code not found in the image.");
         } catch (ChecksumException | FormatException e) {
             System.err.println("Error decoding the QR Code.");
         }
         return null;
     }
 }

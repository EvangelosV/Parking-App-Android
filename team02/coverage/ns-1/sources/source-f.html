 package com.example.parkingspotmanager.domain;
 
 import java.time.Duration;
 import java.time.LocalTime;
 import java.time.YearMonth;
 import java.util.List;
 import java.util.Map;
 import java.util.stream.Collectors;
 
 public class Statistics {
     private List<ParkingSpace> parkingSpaces;
     private List<Payment> payments;
 
     public Statistics(List<ParkingSpace> parkingSpaces, List<Payment> payments) {
         this.parkingSpaces = parkingSpaces;
         this.payments = payments;
     }
 
     // Method to calculate average daily occupancy
     public void calculateAverageDailyOccupancy() {
         parkingSpaces.forEach(space -> {
             double occupancy = space.calculateAverageDailyOccupancy();
             System.out.printf("Parking Space %s Average Daily Occupancy: %.1f%n", space.getParkingSpaceId(), occupancy);
         });
     }
 
     // Method to calculate monthly revenue for a selected year
     public void calculateMonthlyRevenue(int year) {
         Map<Integer, Double> monthlyRevenue = payments.stream()
                 .filter(payment -> payment.getPaymentDate().getYear() == year)
                 .collect(Collectors.groupingBy(payment -> payment.getPaymentDate().getMonthValue(),
                         Collectors.summingDouble(Payment::getAmount)));
 
         monthlyRevenue.forEach((month, revenue) -> {
             System.out.println("Month: " + month + " Revenue: " + revenue);
         });
 
     }
 
     // Method to calculate average parking duration per customer per parking space
     public void calculateAverageParkingDuration() {
         Map<String, Double> averageParkingDuration = payments.stream()
                 .collect(Collectors.groupingBy(
                         payment -> payment.getCustomer().getUsername(),
                         Collectors.averagingDouble(payment -> {
                             if (payment.getReservationTicket() != null) {
                                 return payment.getReservationTicket().getDuration().toHours();
                             } else if (payment.getParkingTicket() != null) {
                                 return payment.getParkingTicket().getDuration().toHours();
                             } else {
                                 return 0.0; // Default value if no ticket exists
                             }
                         })
                 ));
         averageParkingDuration.forEach((customer, duration) -> {
             System.out.println("Customer: " + customer + " Average Parking Duration: " + duration);
         });
 
     }
 }
